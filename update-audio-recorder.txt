CREATE TABLE `sys_audio_records` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT 'id',
  `task_id` bigint NOT NULL COMMENT '任务id',
  `package_id` bigint NOT NULL COMMENT '任务包id',
  `item_order` int NOT NULL COMMENT '在excel中的顺序',
  `text` text NOT NULL COMMENT 'excel中每行文本',
  `modified_text` text COMMENT '修改后的文本',
  `audio_file_path` varchar(500) DEFAULT '' COMMENT '录音文件路径',
  `audio_duration` varchar(10) DEFAULT '00.00',
  `annotator` varchar(20) DEFAULT '' COMMENT '标注员',
  `auditor` varchar(20) DEFAULT '' COMMENT '审计人',
  `create_by` varchar(20) DEFAULT '' COMMENT '创建人',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  `update_by` varchar(20) DEFAULT '' COMMENT '修改人',
  `update_time` datetime DEFAULT NULL COMMENT '修改时间',
  PRIMARY KEY (`id`,`task_id`)
) ENGINE=InnoDB AUTO_INCREMENT=353 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='文件录音表';
